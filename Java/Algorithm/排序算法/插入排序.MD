插入排序
-------
#### Java 实现代码

```java
   public static void main(String[] args) {
        Comparable[] cs = new Integer[]{7, 1, 2, 3, 5, 4, 0, 9, 8, 6};

        System.out.println("排序前：");
        Arrays.stream(cs).forEach(System.out::println);

        insertionSort(cs);

        System.out.println("排序后：");
        Arrays.stream(cs).forEach(System.out::println);
   }

    // 插入排序
    private static void insertionSort(Comparable[] cs) {
        int cl = cs.length;
        for (int i = 1; i < cl; i++) {
            for (int j = i; j > 0 && (cs[j].compareTo(cs[j - 1]) < 0); j--) {// <0 正序； >0 倒序
                Comparable temp=cs[j];
                cs[j]=cs[j-1];
                cs[j-1]=temp;
            }
        }
    }

```
